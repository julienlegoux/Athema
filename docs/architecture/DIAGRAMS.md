## Dependency Injection Container

```typescript
// Container registers implementations
container.register<IConversationProvider>({
  token: 'ConversationProvider',
  useClass: ClaudeAdapter,  // or LocalLLMAdapter
  singleton: true
});

container.register<IRenderer>({
  token: 'Renderer',
  useClass: Live2DRenderer,  // or VRMRenderer
  singleton: true
});

container.register<IAudioEngine>({
  token: 'AudioEngine',
  useClass: ElevenLabsTTS,  // or AzureTTS, LocalTTS
  singleton: true
});

container.register<IMemoryStore>({
  token: 'MemoryStore',
  useClass: SQLiteMemoryStore,
  singleton: true
});

// CharacterController receives dependencies
class CharacterController {
  constructor(
    @inject('Renderer') private renderer: IRenderer,
    @inject('ConversationProvider') private conversation: IConversationProvider,
    @inject('AudioEngine') private audio: IAudioEngine,
    @inject('MemoryStore') private memory: IMemoryStore
  ) {}
}
```

This structure allows swapping any component without changing the core logic.
